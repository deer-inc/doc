Vercel Analytics を使う。

```bash
pnpm add @vercel/analytics
```

```tsx {1, 15} title="app/layout.tsx"
import { Analytics } from '@vercel/analytics/react';

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html>
      <head>
        <title>Next.js</title>
      </head>
      <body>
        {children}
        <Analytics />
      </body>
    </html>
  );
}
```
